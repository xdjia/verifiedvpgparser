Definition cost_eqvv := vpg.cost_eqvv.

Section findRuleWithLc.
  Opaque cost_andb.
  Opaque cost_eqstr.

  Definition findRuleWithLc_sub L i r :=
    match r with
    | (Lr, Alt_Linear i' L') ⇒
      andb (eqvv Lr L) (eqs i i')
    | _ ⇒ false
    end.

  Definition cost_findRuleWithLc_sub_branch1 := 1.
  Definition cost_findRuleWithLc_sub_branch2 := 1.
  Opaque cost_findRuleWithLc_sub_branch1.
  Opaque cost_findRuleWithLc_sub_branch2.

  Definition cost_findRuleWithLc_sub L (r:rule) :=
    match r with
    | (Lr, Alt_Linear _ _) ⇒
      cost_andb + cost_eqstr +
      cost_eqvv Lr L +
      cost_findRuleWithLc_sub_branch1
    | _ ⇒ cost_findRuleWithLc_sub_branch2
    end.

  Program Definition findRuleWithLc_sub' L i (r:rule)
  :{! y !:! _ !<! c !>! c = cost_findRuleWithLc_sub L r ∧
    y = findRuleWithLc_sub L i r !}
  :=
    match r as r' return (r=r')->
      {! y !:! _ !<! c !>!
        c = cost_findRuleWithLc_sub L r ∧
        y = findRuleWithLc_sub L i r !}
    with
    | (Lr, Alt_Linear i' _) ⇒
      fun hyp ⇒
        res <- eqvv' Lr L;
        += cost_andb + cost_eqstr +
        cost_findRuleWithLc_sub_branch1;
        <== andb res (eqs i i')
    | _ ⇒
        fun hyp ⇒
        += cost_findRuleWithLc_sub_branch2;
        <== false
    end (eq_refl r).
  Next Obligation.
    unfold cost_findRuleWithLc_sub.
    split; auto.
    replace vpg.cost_eqvv with cost_eqvv; auto.
    lia.
  Defined.

  Definition cost_findRuleWithLc L i :=
    cost_filter _ (findRuleWithLc_sub L i)
    (cost_findRuleWithLc_sub L) P.

  Program Definition findRuleWithLc' L i
  :{! y !:! _ !<! c !>!
    c = cost_findRuleWithLc L i ∧
    y = findRuleWithLc L i !}
  := filter' _ (findRuleWithLc_sub' L i) P.

End findRuleWithLc.

Section findRuleWithMb.
  Definition findRuleWithMb_sub L a L1 b r :=
    match r with
    | (Lr, Alt_Match i' i2 L' L'') ⇒
      (eqvv Lr L) && (eqvv L' L1) &&
      (String.eqb i' a) && (String.eqb i2 b)
    | _ ⇒ false
    end.

  Definition findRuleWithMb_sub_branch1 := 1.
  Definition findRuleWithMb_sub_branch2 := 1.
  Opaque findRuleWithMb_sub_branch1.
  Opaque findRuleWithMb_sub_branch2.

  Opaque cost_andb.
  Opaque cost_eqstr.

  Definition cost_findRuleWithMb_sub
    L L1 (r:rule) :=
    match r with
    | (Lr, Alt_Match i' i2 L' L'') ⇒
      3 × cost_andb + 2 × cost_eqstr +
      cost_eqvv Lr L + cost_eqvv L' L1 +
      findRuleWithMb_sub_branch1
    | _ ⇒ findRuleWithMb_sub_branch2
    end.

  Program Definition findRuleWithMb_sub'
    L a L1 b (r:rule)
  :{! y !:! _ !<! c !>! c = cost_findRuleWithMb_sub L L1 r ∧
    y = findRuleWithMb_sub L a L1 b r !}
  :=
    match r as r' return (r=r')->
      {! y !:! _ !<! c !>!
        c = cost_findRuleWithMb_sub L L1 r ∧
        y = findRuleWithMb_sub L a L1 b r !}
    with
    | (Lr, Alt_Match i' i2 L' L'') ⇒
      fun hyp ⇒
        res1 <- eqvv' Lr L;
        res2 <- eqvv' L' L1;
        += 3 × cost_andb + 2 × cost_eqstr +
        findRuleWithMb_sub_branch1;
        <== res1 && res2 && (String.eqb i' a) && (String.eqb i2 b)
    | _ ⇒
      fun hyp ⇒
      += findRuleWithMb_sub_branch2;
      <== false
    end (eq_refl r).

  Definition cost_findRuleWithMb L a L1 b :=
    cost_filter _ (findRuleWithMb_sub L a L1 b)
    (cost_findRuleWithMb_sub L L1) P.

  Next Obligation.
    split; auto.
    unfold cost_findRuleWithMb_sub.
    replace vpg.cost_eqvv with cost_eqvv; auto.

    lia.
  Defined.

  Program Definition findRuleWithMb' L a L1 b
  :{! y !:! _ !<! c !>!
    c = cost_findRuleWithMb L a L1 b ∧
    y = findRuleWithMb L a L1 b !}
  :=
    filter' _
    (findRuleWithMb_sub' L a L1 b) P.
End findRuleWithMb.

Section convert2PlainE.

    end.

    unfold cost_convert2PlainE; auto.
  Qed.

End convert2PlainE.

Section fold'.
  Variable A B : Type.

  Definition cost_fold_branch1 := 1.
  Definition cost_fold_branch2 := 1.
  Opaque cost_fold_branch1.
  Opaque cost_fold_branch2.

  Fixpoint cost_fold
    {f:A→B→A}
    {cost_f:A→B→nat} l a :=
    match l with
    | nil ⇒ cost_fold_branch1
    | cons b t ⇒
      cost_f a b +
      @cost_fold f cost_f t (f a b) +
      cost_fold_branch2
    end.

  Program Fixpoint fold'
    {f:A→B→A}
    {cost_f:A→B→nat}
    (f':∀ a b,
      {! y !:! A !<! c !>! c = cost_f a b ∧
        y = f a b !})
    l a
  :{! y !:! _ !<! c !>! c = @cost_fold f cost_f l a ∧
    y = fold_left f l a !}
  :=
  match l as l' return (l=l')->
  {! y !:! _ !<! c !>! c = @cost_fold f cost_f l a ∧
    y = fold_left f l a !}
  with
  | nil ⇒
    fun hyp ⇒
    += cost_fold_branch1; <== a
  | cons b t ⇒
    fun hyp ⇒
    fab <- f' a b;
    out <- @fold' f cost_f f' t fab;
    += cost_fold_branch2; <== out
  end (eq_refl l).

  Next Obligation.
    simpl. split; auto. lia.
  Qed.
End fold'.
